import socket
import os

UDP_IP = "0.0.0.0"       # Listen on all network interfaces
UDP_PORT = 5001
BUFFER_SIZE = 1024

# Create UDP socket
server_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
server_socket.bind((UDP_IP, UDP_PORT))

print("UDP Server starting...")
print(f"UDP Server is listening on port {UDP_PORT}...")

# Receive filename first
data, addr = server_socket.recvfrom(BUFFER_SIZE)
filename = data.decode()
filename = os.path.basename(filename)       # remove path if sent
save_as = "received_" + filename
print(f"Receiving file: {filename} from {addr}")

with open(save_as, "wb") as f:
    chunk_count = 0
    while True:
        data, addr = server_socket.recvfrom(BUFFER_SIZE)
        if data == b'EOF':                   # EOF marker from client
            print("End of file reached.")
            break
        f.write(data)
        chunk_count += 1
        print(f"Received chunk #{chunk_count}, {len(data)} bytes")

print(f"\nFile received successfully as '{save_as}'")
server_socket.close()

import socket
import os

UDP_IP = input("Enter server IP address: ")
UDP_PORT = 5001
BUFFER_SIZE = 1024

filename = input("Enter the file name to send: ")
basename = os.path.basename(filename)  # just the file name

client_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

# Send filename first
client_socket.sendto(basename.encode(), (UDP_IP, UDP_PORT))
print(f"Filename '{basename}' sent to server.")

# Send file data
with open(filename, "rb") as f:
    chunk_count = 0
    while True:
        chunk = f.read(BUFFER_SIZE)
        if not chunk:
            break
        client_socket.sendto(chunk, (UDP_IP, UDP_PORT))
        chunk_count += 1
        print(f"Sent chunk #{chunk_count}, {len(chunk)} bytes")

# Send EOF marker
client_socket.sendto(b'EOF', (UDP_IP, UDP_PORT))
print("File sent successfully.")

client_socket.close()
